pacman -Syy 


@Disk partition@

fdisk /dev/sda - MBR

cgdisk /dev/sda - GPT

# /boot 100M - выставить флаг boot
# / 20G
# swap 1024M
# /home весь остаток

#(MBR)                          #(GPT) 
mkfs.ext2  /dev/sda1 -L boot  mkfs.fat -F32 /dev/sda3 (boot)
mkfs.ext4  /dev/sda2 -L root   mkfs.ext4 /dev/sda1
mkswap /dev/sda3 -L swap 
mkfs.ext4  /dev/sda4 -L home mkfs.ext4 /dev/sda2 (home)

@mounting disks@

mount /dev/sda2 /mnt
mkdir /mnt/{boot,home}
mount /dev/sda1 /mnt/boot
swapon /dev/sda3
mount /dev/sda4 /mnt/home

@Install base system@
pacstrap -i /mnt base base-devel

@system settings fstab generating@

genfstab -U -p /mnt >> /mnt/etc/fstab

@load to a new system@
arch-chroot /mnt

@PC name@
nano /etc/hostname

@Time settings@
ln -S /usr/share/zoneinfo/Europe/Moscow /etc/localtime

@Add locale@
nano /etc/locale.gen
locale-gen

@system language@
nano /etc/locale.conf
LANG="en_US.UTF-8"

@console keymap@
nano /etc/vconsole.conf
KEYMAP=ru
FONT=cyr-sun16

@Create RAM disk@
mkinitcpio -p linux

@set root password@
passwd

@Boot loader installation@
#UEFI 
mount -t efivarfs efivarfs /sys/firmware/efi/efivars
bootctl install
blkid -s PARTUUID -o value /dev/sda3(root)
nano /boot/loader/entries/arch.conf
#example 
title Arch Linux
linux /vmlinuz-linux
initrd /initramfs-linux.img
options root=PARTUUID=84c7e76f-43f6-4509-9b02-f3f5c0eac3b2_rw

#Grub
pacman - S grub
#if will be two OS in one HDD, so:
 pacman - S os-prober mtools fuse

@install bootloader to the disk@
grub-install /dev/sda
grub-mkconfig -o /boot/grub/grub.cfg

@wifi module@
pacman - S dialog wpa_supplicant

exit

@unmounting parts@
umount /mnt/{boot,home}
reboot
